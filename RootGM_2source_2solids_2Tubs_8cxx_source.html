95double RootGM::Tubs::StartPhi() const
96{
97 if (fTubs->TestShapeBit(TGeoShape::kGeoTubeSeg))
98 return ((TGeoTubeSeg*)fTubs)->GetPhi1() * RootGM::Units::Angle();
99 else
100 return 0. * RootGM::Units::Angle();
101}
102
103//_____________________________________________________________________________
104double RootGM::Tubs::DeltaPhi() const
105{
106 if (fTubs->TestShapeBit(TGeoShape::kGeoTubeSeg)) {
107 return (((TGeoTubeSeg*)fTubs)->GetPhi2() -
108 ((TGeoTubeSeg*)fTubs)->GetPhi1()) *
109 RootGM::Units::Angle();
110 }
111 else
112 return 360. * RootGM::Units::Angle();
113}
