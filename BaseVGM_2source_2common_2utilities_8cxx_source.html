34double BaseVGM::Round(double x)
35{
38
39 double t;
40 if (x >= 0.0) {
41 t = ceil(x);
42 if (t - x > 0.5) t -= 1.0;
43 return t;
44 }
45 else {
46 t = ceil(-x);
47 if (t + x > 0.5) t -= 1.0;
48 return -t;
49 }
50}
