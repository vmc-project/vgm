25std::ostream& operator<<(std::ostream& out, const VGM::IVolume& volume)
26{
27 out << "Volume: "
28 << "\"" << volume.Name() << "\""
29 << " material "
30 << "\"" << volume.MaterialName() << "\""

33 << volume.NofDaughters() << " daughters: ";
