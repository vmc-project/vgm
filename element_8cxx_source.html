22std::ostream& operator<<(std::ostream& out, const VGM::IElement& element)
23{
24 out << "Element: "

26 << "\"" << element.Symbol() << "\""

28 << " A = " << element.A() << " g/mole";
29
30 for (int i = 0; i < element.NofIsotopes(); i++) {
31 if (i == 0) out << std::endl;

34 if (i < element.NofIsotopes() - 1) out << std::endl;
35 }
